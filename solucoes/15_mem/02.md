short *obter_sinal(double *array, int tamanho) 
{
	short sArray[tamanho];
    for( int i = 0; i < tamanho; i++)
    {
	    if(array[i] > 0)
	    {
		    sArray[i] = 1;
	    }
	    else if( array[i] < 0)
	    {
			sArray[i] = -1;
		}
		else
		{
			sArray[i] = 0;
		}
    }
    
    return sArray;
}
